153 esp_err_t _gn_init_spiffs(gn_config_handle_intl_t conf) {
154 
155  ESP_LOGD(TAG, "_gn_init_spiffs");
156 
157  esp_vfs_spiffs_conf_t vfs_conf;
158  vfs_conf.base_path = "/spiffs";
159  vfs_conf.partition_label = NULL;
160  vfs_conf.max_files = 6;
161  vfs_conf.format_if_mount_failed = true;
162 
163  conf->spiffs_conf = vfs_conf;
164 
165  // Use settings defined above to initialize and mount SPIFFS filesystem.
166  // Note: esp_vfs_spiffs_register is anall-in-one convenience function.
167  esp_err_t ret = ESP_OK;
168 
169  ret = esp_vfs_spiffs_register(&vfs_conf);
170 
171  if (ret != ESP_OK) {
172  if (ret == ESP_FAIL) {
173  gn_log(TAG, GN_LOG_ERROR, "Failed to mount or format filesystem");
174  } else if (ret == ESP_ERR_NOT_FOUND) {
175  gn_log(TAG, GN_LOG_ERROR, "Failed to find SPIFFS partition");
176  } else {
177  gn_log(TAG, GN_LOG_ERROR, "Failed to initialize SPIFFS (%s)",
178  esp_err_to_name(ret));
179  }
180  return ret;
181  }
182 
183  size_t total = 0, used = 0;
184  ret = esp_spiffs_info(NULL, &total, &used);
185  if (ret != ESP_OK) {
186  gn_log(TAG, GN_LOG_ERROR,
187  "Failed to get SPIFFS partition information (%s)",
188  esp_err_to_name(ret));
189  } else {
190  ESP_LOGD(TAG, "Partition size: total: %d, used: %d", total, used);
191  }
192 
193  return ret;
194
